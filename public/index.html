# Day 1: Frontier model cost comparison
models = {
    "Claude 3.5 Sonnet": {"ctx":"200K","in":3.0,"out":15.0},
    "GPT-4o":            {"ctx":"128K","in":5.0,"out":15.0},
    "Gemini 1.5 Pro":    {"ctx":"1M",  "in":3.5,"out":10.5},
    "o3":                {"ctx":"128K","in":10.0,"out":40.0},
}
for name, s in models.items():
    cost = (500/1e6*s['in'] + 300/1e6*s['out']) * 1000
    print(f"{name:<22}  ${cost:.4f}")
